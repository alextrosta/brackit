\section*{Functional Requirements (FR)}

\begin{itemize}

    \item \textbf{\textit{Bracket Generation/Maintenance}}
        \begin{todolist}
            \item[\done]{{TOs can create joinable double elimination brackets}}\label{1.1}
            \item[\done]{{Entrants can use system to enter joinable double elimination brackets}}\label{1.2}
            \item[\done]{{Given a seeded list of entrants, output a correct double elimination bracket}}\label{1.3}
            \item[\done]{{TOs can access a setup screen where they seed (rank) the entrants dynamically and can view a preview of the bracket as seeded, then confirm the final active bracket}}\label{1.4}
            
                \begin{itemize}
                    \item{If output of system is not accepted, TOs can reseed entrants}
                \end{itemize}
        \end{todolist}

    \item \textbf{\textit{Profile Creation/Maintenance}}
        \begin{todolist}
            \item[\done]{Competitors/Organizers can create profiles which keep a history of:}\label{2.1}
            \begin{itemize}
                \item{tournaments entered and created, placement history}
                \item{number of matches won/lost}
                \item{overall matchups against opponents}
                \item{additional user profile information}
            \end{itemize}
            \item[\done]{The system supports both user profiles and competitors who haven’t created a profile (Guest Users)}
        \end{todolist}

    \item \textbf{\textit{Data Input/Processing}}
        \begin{todolist}
            \item[\done]{Tournament organizers can enter tournament results in active brackets}
            \item[\done]{Use dynamic tournament results to render Losers brackets and subsequent Winners rounds during competitions in real time}

        \end{todolist}

    \item \textbf{\textit{Bracket Visualization}}
        \begin{todolist}
            \item[\done]{Users can access multiple views of brackets:} \label{4.1}
            \begin{itemize}
                \item{Winners / Losers brackets}
                \item{Specific bracket rounds}
            \end{itemize}
            \item[\partly]{Users can click on competitors in rounds to view their profile}
        \end{todolist}
    
\end{itemize}
\section*{Non-Functional Requirements (NFR)}
\begin{itemize}
\item \textbf{\textit{Useability}}: \texttt{Brackit's} main selling point - current platforms are simply not usable on mobile devices. 
    \begin{todolist}
    \item[\done]{Users can clearly view and understand bracket structure and competitors’ placement within brackets}
    \item[\done]{Visually coherent mobile representation of a double elimination bracket}
    \item[\partly]{Seamless competitor entry (by TOs) to brackets in the setup phase}
    \end{todolist}

\item \textbf{\textit{Screen Adaption}}
    \begin{todolist}
    \item[\partly]{Users can visualize brackets in portrait and landscape modes}
    \end{todolist}
